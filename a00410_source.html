136  void compute_edge_point(const T& _v_1, const T& _v_2, int e, const pnt_type& p, const vec_type& d,
137  pnt_type& q, vec_type& n)
138  {
139  X de = d(e);
140  T v_1 = _v_1;
141  T v_2 = _v_2;
142  pnt_type p_end = p;
143  q = p;
144  unsigned int nr_iters = 0;
145  bool finished;
146  X ref = consistency_threshold*fabs(v_2-v_1);
147  do {
148  finished = false;
149  // compute point on edge
150  X alpha;
151  if (fabs(v_2-v_1) > epsilon) {
152  finished = true;
153  alpha = (X)(iso_value-v_1)/(v_2-v_1);
154  }
155  else
156  alpha = (X) 0.5;
157 
158  q(e) = p_end(e) - (1-alpha)*de;
159  // compute normal at point
160  n = vec_type(*func.evaluate_gradient(q.to_vec()));
161 
162  // stop if maximum number of iterations reached
163  if (++nr_iters >= max_nr_iters)
164  break;
165  // check if gradient is in accordance with value difference
166  if (finished) {
167  if (alpha < 0 || alpha > 1) {
168  std::cout << "invalid alpha = " << alpha << std::endl;
169  }
170  X f_e = (v_2 - v_1) / de;
171  if (fabs(f_e - n(e)) > consistency_threshold*f_e) {
172  X v = func.evaluate(q.to_vec());
173  if (fabs(v-iso_value) > ref) {
174  if ((v > iso_value) == (v_2 > iso_value)) {
175  de *= alpha;
176  v_2 = v;
177  p_end(e) = q(e);
178  }
179  else {
180  v_1 = v;
181  de *= 1-alpha;
182  }
183  finished = false;
184  }
185  }
186  }
187  } while (!finished);
188  if (nr_iters > 15)
189  std::cout << "not converged after " << nr_iters << " iterations" << std::endl;
190  // if gradient does not define normal
191  if (n.length() < 1e-6) {
192  // define normal from edge direction
193  n = vec_type(0, 0, 0);
194  n(e) = T((v_1 < v_2) ? 1 : 0);
195  }
196  else
197  n.normalize();
198  }

289  void extract(const T& _iso_value,
290  const axis_aligned_box<X,3>& box,
291  unsigned int _resx, unsigned int _resy, unsigned int _resz,
292  bool show_progress = false)
293  {
294  // prepare private members
295  resx = _resx; resy = _resy; resz = _resz;
296  minp = p = box.get_min_pnt();
297  d = box.get_extent();
298  d(0) /= (resx-1); d(1) /= (resy-1); d(2) /= (resz-1);
299  iso_value = _iso_value;
300 
301  // prepare progression
302  cgv::utils::progression prog;
303  if (show_progress) prog.init("extraction", resz, 10);
304 
305  // construct three slice infos
306  dc_slice_info<T> slice_info_1(resx,resy), slice_info_2(resx,resy), slice_info_3(resx,resy);
307  dc_slice_info<T> *slice_info_ptrs[3] = { &slice_info_1, &slice_info_2, &slice_info_3 };
308 
309  // iterate through all slices
310  unsigned int nr_vertices[4] = { 0, 0, 0, 0 };
311  unsigned int k, n;
312 
313  process_slice(0, slice_info_ptrs[0]);
314  p(2) += d(2);
315  process_slice(slice_info_ptrs[0], slice_info_ptrs[1]);
316  process_slab(slice_info_ptrs[0], slice_info_ptrs[1]);
317  p(2) += d(2);
318  // show progression
319  if (show_progress) {
320  prog.step();
321  prog.step();
322  }
323  for (k=2; k<resz; ++k, p(2) += d(2)) {
324  n = base_type::get_nr_vertices();
325  // evaluate function on next slice and construct slice interior vertices
326  dc_slice_info<T> *info_ptr_0 = slice_info_ptrs[(k-2)%3];
327  dc_slice_info<T> *info_ptr_1 = slice_info_ptrs[(k-1)%3];
328  dc_slice_info<T> *info_ptr_2 = slice_info_ptrs[k%3];
329  process_slice(info_ptr_1, info_ptr_2);
330  process_slab(info_ptr_1, info_ptr_2);
331  generate_slice_quads(info_ptr_0, info_ptr_1);
332 
333  n = base_type::get_nr_vertices()-n;
334  nr_vertices[k%4] = n;
335  n = nr_vertices[(k+3)%4];
336  if (n > 0)
337  base_type::drop_vertices(n);
338  // show progression
339  if (show_progress)
340  prog.step();
341  }
