94  RPF_DRAWABLES_INIT_FRAME = 1 << 10, // whether to call the init_frame method of the drawables
95  RPF_SET_STATE_FLAGS = 1 << 11, // whether to set depth buffer and culling flags
96  RPF_SET_CLEAR_COLOR = 1 << 12, // whether to set the clear color
97  RPF_SET_CLEAR_DEPTH = 1 << 13, // whether to set the clear color
98  RPF_SET_CLEAR_STENCIL = 1 << 14, // whether to set the clear color
99  RPF_SET_CLEAR_ACCUM = 1 << 15, // whether to set the accumulation buffer clear color
100  RPF_DRAWABLES_DRAW = 1 << 16, // whether to call draw and finish_draw methods of drawables
101  RPF_DRAWABLES_FINISH_FRAME = 1 << 17, // whether to call finish frame method of drawables
102  RPF_DRAW_TEXTUAL_INFO = 1 << 18, // whether to draw textual information
103  RPF_DRAWABLES_AFTER_FINISH = 1 << 19, // whether to call after finish method of drawables
