68  look_at4(const fvec<T, 3>& eye, const fvec<T, 3>& focus, const fvec<T, 3>& view_up_dir) {
69  fvec<T, 3> view_dir = focus-eye;
70  view_dir.normalize();
71  fvec<T, 3> up_dir(view_up_dir);
72  up_dir.normalize();

74  x.normalize();
75  fvec<T, 3> y = cross(x, view_dir);
76  fmat<T, 4, 4> R = identity4<T>();
77  R(0, 0) = x(0); R(0, 1) = x(1); R(0, 2) = x(2);
78  R(1, 0) = y(0); R(1, 1) = y(1); R(1, 2) = y(2);
79  R(2, 0) = -view_dir(0); R(2, 1) = -view_dir(1); R(2, 2) = -view_dir(2);
80  return R*translate4<T>(-eye);

113  template <typename T> fmat<T, 4, 4>
114  perspective4(const T& fovy, const T& aspect, const T& zNear, const T& zFar) {
115  fmat<T, 4, 4> M;
116  M.zeros();
117  T angle = T(0.008726646262)*fovy;
118  T f = cos(angle)/sin(angle);
119  M(0, 0) = f / aspect;
120  M(1, 1) = f;
121  M(2, 2) = (zNear + zFar) / (zNear - zFar);
122  M(2, 3) = T(2) * zNear * zFar / (zNear - zFar);
123  M(3, 2) = -T(1);
124  return M;
125  }
